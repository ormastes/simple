# PROPER CMakeLists.txt for Module 59 that correctly uses Module 58

add_library(attention_expert_kernels_59 STATIC
    kernels/attention_fwd.cu
    kernels/expert_fwd.cu
)

target_include_directories(attention_expert_kernels_59
    PUBLIC
        ${CMAKE_CURRENT_SOURCE_DIR}/common
)

target_link_libraries(attention_expert_kernels_59
    PUBLIC
        ${CUDA_BASIC_LIB}
)

target_compile_features(attention_expert_kernels_59 PUBLIC cxx_std_20)

find_package(Torch QUIET)

if(Torch_FOUND)
    # PROPER implementation using Module 58
    add_library(attention_expert_torch_proper_59 SHARED
        pytorch/attention_pytorch.cpp
        pytorch/nvme_tensor_loader_proper.cpp  # Uses Module 58 properly
    )

    target_include_directories(attention_expert_torch_proper_59
        PUBLIC
            ${CMAKE_CURRENT_SOURCE_DIR}/common
            ${TORCH_INCLUDE_DIRS}
            # Include Module 58 headers
            ${CMAKE_SOURCE_DIR}/50.GPU-NVMe_Interaction/58.Simple_GPU_Filesystem_API/src/common
            # Include Module 53 headers (for device management)
            ${CMAKE_SOURCE_DIR}/50.GPU-NVMe_Interaction/53.NVMe_VFIO_Host_Layer/src/common
    )

    target_link_libraries(attention_expert_torch_proper_59
        PUBLIC
            attention_expert_kernels_59
            Torch::Torch
            ${CUDA_BASIC_LIB}
            # PROPERLY LINK TO MODULE 58!
            58_Simple_Filesystem_Layer_lib
            # Also need Module 53 for device management
            nvme_vio_host
            host_io_host_mem
            device_detector
    )

    target_compile_features(attention_expert_torch_proper_59 PRIVATE cxx_std_20)
    set_target_properties(attention_expert_torch_proper_59 PROPERTIES
        OUTPUT_NAME attention_expert_torch_proper_59
    )

    # Original (incorrect) implementation for comparison
    add_library(attention_expert_torch_59 SHARED
        pytorch/attention_pytorch.cpp
        pytorch/nvme_tensor_loader.cpp  # Bad: uses direct POSIX I/O
    )

    target_include_directories(attention_expert_torch_59
        PUBLIC
            ${CMAKE_CURRENT_SOURCE_DIR}/common
            ${TORCH_INCLUDE_DIRS}
    )

    target_link_libraries(attention_expert_torch_59
        PUBLIC
            attention_expert_kernels_59
            Torch::Torch
            ${CUDA_BASIC_LIB}
            # NOTE: Original doesn't link to Module 58!
    )

    target_compile_features(attention_expert_torch_59 PRIVATE cxx_std_20)
    set_target_properties(attention_expert_torch_59 PROPERTIES
        OUTPUT_NAME attention_expert_torch_59
    )

    set(ATTENTION_EXPERT_TORCH_FOUND TRUE PARENT_SCOPE)

    message(STATUS "Module 59: Building both implementations:")
    message(STATUS "  - attention_expert_torch_59 (original - uses direct I/O)")
    message(STATUS "  - attention_expert_torch_proper_59 (proper - uses Module 58)")
else()
    message(WARNING "PyTorch not found - skipping Module 59 bindings")
    set(ATTENTION_EXPERT_TORCH_FOUND FALSE PARENT_SCOPE)
endif()
