```simple
step("Load the checked-in cross-language smoke report")
val report = rt_file_read_text("doc/09_report/cross_language_perf_parallel_smoke.md") ?? ""
step("Select the OS-thread parallel worker section")
val parallel = section_named(report, "OS Thread Parallel Workers")

step("Read Simple, Go, and C worker timings")
val simple_os = row_ms_scaled(parallel, "Simple (native)")
val multicore = row_ms_scaled(parallel, "Simple multicore green (native)")
val go_ms = row_ms_scaled(parallel, "Go")
val c_ms = row_ms_scaled(parallel, "C (pthreads)")
val native_baseline = min_positive(go_ms, c_ms)

step("Verify all worker rows have positive timing evidence")
expect(simple_os).to_be_greater_than(0)
expect(multicore).to_be_greater_than(0)
expect(go_ms).to_be_greater_than(0)
expect(c_ms).to_be_greater_than(0)
step("Verify Simple worker rows stay inside smoke thresholds")
expect(simple_os).to_be_less_than(native_baseline * 2 + 1)
expect(multicore).to_be_less_than(native_baseline * 3 + 1)
step("Verify concurrency model labels preserve semantic distinctions")
expect(model_text(row_for_label(parallel, "Simple cooperative green (native)"))).to_contain("cooperative")
expect(model_text(row_for_label(parallel, "Simple multicore green (native)"))).to_contain("parallelism=")
expect(report).to_contain("not a Go M:N goroutine equivalent")
expect(report).to_contain("used_runtime_pool()")
```

```simple
step("Load the checked-in cross-language smoke report")
val report = rt_file_read_text("doc/09_report/cross_language_perf_parallel_smoke.md") ?? ""
step("Select the Simple versus Go versus C stress section")
val stress = section_named(report, "Simple vs Go vs C Large Fanout Stress")

step("Read stress fanout timings")
val simple_multicore = row_ms_scaled(stress, "Simple multicore green (native)")
val go_ms = row_ms_scaled(stress, "Go")
val c_ms = row_ms_scaled(stress, "C (pthreads)")

step("Verify stress rows have positive timing evidence")
expect(simple_multicore).to_be_greater_than(0)
expect(go_ms).to_be_greater_than(0)
expect(c_ms).to_be_greater_than(0)
step("Verify Go beats or matches C pthread stress fanout")
expect(go_ms).to_be_less_than(c_ms + 1)
step("Verify stress row model labels prove runtime-pool and queue evidence")
expect(model_text(row_for_label(stress, "Simple multicore green (native)"))).to_contain("multicore_green stress fanout")
expect(model_text(row_for_label(stress, "Simple multicore green (native)"))).to_contain("pool_used=")
expect(model_text(row_for_label(stress, "Simple multicore green (native)"))).to_contain("parallelism=")
expect(model_text(row_for_label(stress, "Simple multicore green (native)"))).to_contain("queue_model=work_stealing")
expect(model_text(row_for_label(stress, "Go"))).to_contain("goroutine per stress task")
expect(model_text(row_for_label(stress, "C (pthreads)"))).to_contain("pthread per stress task")
```
