```simple
# Exercises pos-slot >= 14 distance decode (direct + align bits) which
# is where the u32 wrap-around bug originally surfaced before the
# range decoder was ported.
_ensure_tmp_root()
val input = _payload(256)
val input_path = TMP_ROOT + "/p256.bin"
val xz_path = TMP_ROOT + "/p256.xz"
_write_bytes_via_hex_shell(input_path, input)
val run = shell("xz -q -z -f --check=crc32 -c '" + input_path + "' > '" + xz_path + "'")
expect(run.exit_code).to_equal(0)
val encoded = rt_file_read_bytes(xz_path) ?? []
val decoded = lzma2_decompress_xz(encoded)
expect(decoded.is_err()).to_equal(false)
expect(decoded.unwrap()).to_equal(input)
```

```simple
# Host `xz` defaults to CRC64. Supporting that check closes the main
# remaining interop gap for single-filter `.xz` streams without
# changing the LZMA2 payload decoder.
_ensure_tmp_root()
val input = _payload(64)
val input_path = TMP_ROOT + "/crc64.bin"
val xz_path = TMP_ROOT + "/crc64.xz"
_write_bytes_via_hex_shell(input_path, input)
val run = shell("xz -q -z -f -c '" + input_path + "' > '" + xz_path + "'")
expect(run.exit_code).to_equal(0)
val encoded = rt_file_read_bytes(xz_path) ?? []
val decoded = lzma2_decompress_xz(encoded)
expect(decoded.is_err()).to_equal(false)
```
