
### Encoding/Decoding
```simple
use std.encoding_utils

# Base64

val decoded = base64_decode_string(encoded)  # "Hello World"

# Hexadecimal
val hex = hex_encode_string("Hello")  # "48656c6c6f"
val text = hex_decode_string(hex)  # "Hello"

# URL encoding
val url = url_encode("hello world!")  # "hello%20world%21"
val original = url_decode(url)  # "hello world!"

# ROT13
val encrypted = rot13_encode("secret")  # "frperg"
val decrypted = rot13_decode(encrypted)  # "secret"
```
